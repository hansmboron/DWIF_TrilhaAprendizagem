Introdução JavaScript

JavaScript foi criada em 1995 pela Netscape em parceria com a Sun Mycrosystem com a finalidade de adicionar interatividade a uma página web.

Inserção do JavaScript no HTML

Incorporado:


<head>
    <script rel="text/javascript">
        comandos
    </script>
</head>

Externo:


<script rel="text/javascript" src="externo.js"></script>

Atributo de Evento:


<body>
    <button type="button" onclick="comandos"></button>
</body>

Estruturas condicionais


/* Elabore um programa em que solicite ao usuário dois números n1 e n2.
Realize a subtração dos dois valores. O programa não deve exibir números negativos */

var n1, n2, subtracao
n1 = parseInt(prompt("Digite primeiro valor: "))
n2 = parseInt(prompt("Digite segundo valor: "))
if (n1 > n2) {
    subtracao = n1 - n2
    alert("O resultado é: " + subtracao)
} else {
    subtracao = n2 - n1
    alert("O resultado é: " + subtracao)
}

Laços de Repetição

While:


/* Permita ao usuário inserir um valor que exiba a tabuada */

var tab, num, count = 0
num = parseInt(prompt("Digite o valor da tabuada: "))

while(count < 11) {
    tab = num * count
    document.write("<h2>" + tab + "</h2>")
    count ++
}

For:


/* Permita ao usuário inserir um valor que exiba a tabuada */

var tab, num, count
num = parseInt(prompt("Digite o valor da tabuada: "))

for(count = 0; count < 11; count ++) {
    tab = num * count
    document.write("<h2>" + tab + "</h2>")
}

Estruturas de Multipla Escolha

Switch:


/* Permita ao usuário inserir um valor que exiba a tabuada */

var tab, num, count = 0, op
num = parseInt(prompt("Digite o valor da tabuada: "))
op = parseInt(prompt("Digite 1 para While ou 2 para For: "))

switch(op) {
    case 1: {
        while(count < 11) {
            tab = num * count
            document.write("<h2>" + tab + "</h2>")
            count ++
        }
        break
    } 
    case 2: {
        for(count = 0; count < 11; count ++) {
            tab = num * count
            document.write("<h2>" + tab + "</h2>")
        }
        break
    }
    default: {
        alert("Opção Inválida!")
        break
    }
}

Funções

Funções sem retorno:


function somar(n1, n2) {
    var r = n1 + n2
    alert("Resultado: " + r)
}

somar(35, 10)  /* 'Resultado: 45' */
somar(212, -14)  /* 'Resultado: 198' */

Funções com retorno:


function subtrair(n1, n2) {
    var r = n1 - n2
    return r
}

alert(subtrair(35, 10))  /* 25 */
alert(subtrair(212, -14))  /* 226 */
